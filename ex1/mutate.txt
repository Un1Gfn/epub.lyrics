infinite loop
  read uCur from stdin
  switch uCur
    case EOF
      emit(uPrev, uPrevM)
      print sNonHangul
      break
    case NonHangul
      sNonHangul += uCur
    case Hangul
      if uPrev nil # beginning of file
        uPrev, uPrevM <- uCur, uCur
        continue
      else
        uPrevMM, uCurM <- mutate2(uPrevM, mutate1(uCur))
        emit(uPrev, uPrevMM)
        print sNonHangul
        sNonHangul, uPrev, uPrevM <- "", uCur, uCurM
